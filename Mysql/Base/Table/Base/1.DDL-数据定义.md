## DDL - 数据定义语言
```
DDL（Data Definition Language）数据定义语言
定义/修改数据库对象结构：库、表、列、索引、视图等。
常见语句：
CREATE（建库/建表/建索引/建视图）
ALTER（改表结构）
DROP（删表/删库）
TRUNCATE（清空表，重置存储结构，通常比 DELETE 更“硬”）
```

### 1. 表操作
```
# 如果存在则删除
drop table if exists 表名;
# 创建
create table [IF NOT EXISTS] 表名(
    字段1 字段类型[comment 注释],
    字段2 字段类型[comment 注释],
    ...
    字段n 字段类型[comment 注释]
)[comment 表注释];

# 查询
show tables;                          # 展示所有
desc 表名;                            # 查看表结构
show create table 表名;               # 查看指定表的建表语句

# 删除，注意该删除方法属于DDL，无法回滚
drop table 表名;                      # 删除表
truncate table 表名;                  # 清空表，重置存储结构

# 修改
alter tabel 旧表名 rename to 新表名;
```

### 2. 字段操作
```
# 添加
alter table 表名 add [column] 字段名 数据类型;                   # 添加到最后面
alter table 表名 add [column] 字段名 数据类型 after 字段;         # 指定在字段后追加新字段
ALTER TABLE 表名 ADD [column] 字段名 数据类型 FIRST;             # 添加到最前面

# 删除
alter table 表名 drop 字段名;

# 修改
alter table 表名 modify 字段名 新数据类型;                       # 修改字段类型
alter table 表名 change 字段名 新字段名 新数据类型;               # 修改字段名称和字段类型
```
    
### 3. 其他操作
```
# 数据表拷贝（结构）
create table table1 as select * from table2 where 1=2;  # 不会拷贝索引、主键、约束、默认值
create table table1 like table2;  # 适用于mysql

# 数据表拷贝 （结构+数据）
create table table1 as select * from table2;  # 不包含索引、约束、触发器

# 插入数据
insert into table1 select * from table2;
```
